function login() { const ic = document.getElementById("ic").value.trim(); const pw = document.getElementById("password").value; if (!ic || !pw) { alert("Sila isi semua maklumat."); return; } const userData = JSON.parse(localStorage.getItem("users")) || []; const user = userData.find(u => u.ic === ic && u.password === pw); if (user) { localStorage.setItem("currentUser", JSON.stringify(user)); window.location.href = "homepage.html"; } else { alert("Log masuk gagal. Sila cuba lagi."); } }
